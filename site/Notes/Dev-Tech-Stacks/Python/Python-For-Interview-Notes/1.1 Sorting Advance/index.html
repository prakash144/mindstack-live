    🎯 Custom Sort Transformations¶

    1️⃣ Sort by Calculated Value¶

    # Example: Sort numbers by their square value difference
# LeetCode similar: "Minimum Distance Between Sorted Numbers"
nums = [5, -2, 3, -4, 1]
sorted_nums = sorted(nums, key=lambda x: x*x)  # Sort by square value
# [-2, 1, 3, -4, 5]

    2️⃣ Sort with Custom Comparison¶

    # Example: Sort strings by custom order
# LeetCode similar: "Custom Sort String"
order = "cba"
words = ["apple", "banana", "cherry"]
sorted_words = sorted(words, key=lambda x: [order.index(c) if c in order else len(order) for c in x])

    🔄 Multi-Key Sorting¶

    1️⃣ Tuple-based Sorting¶

    # Example: Sort intervals by start, then end
# LeetCode similar: "Merge Intervals"
intervals = [[1,3], [2,6], [8,10], [15,18]]
intervals.sort(key=lambda x: (x[0], x[1]))

# Sort logs with multiple conditions
# LeetCode similar: "Reorder Data in Log Files"
logs = ["dig1 8 1 5 1", "let1 art can", "dig2 3 6", "let2 own kit dig", "let3 art zero"]
logs.sort(key=lambda x: (0 if x.split()[1].isalpha() else 1,  # letters before digits
                        ' '.join(x.split()[1:]) if x.split()[1].isalpha() else None,  # content
                        x.split()[0] if x.split()[1].isalpha() else None))  # identifier

    🎲 Object Sorting¶

    1️⃣ Custom Class Sorting¶

    # Example: Sort meetings by complexity
# LeetCode similar: "Meeting Rooms III"
class Meeting:
    def __init__(self, start, end, complexity):
        self.start = start
        self.end = end
        self.complexity = complexity

meetings = [Meeting(1, 3, 2), Meeting(2, 4, 1), Meeting(3, 5, 3)]
meetings.sort(key=lambda x: (x.complexity, x.start, x.end))

    📊 Bucket Sort Pattern¶

    1️⃣ Frequency Sort¶

    # Example: Sort characters by frequency
# LeetCode similar: "Sort Characters By Frequency"
from collections import Counter

def frequencySort(s: str) -> str:
    # Count frequencies
    freq = Counter(s)
    # Sort characters by frequency, then by character value
    return ''.join(sorted(s, key=lambda x: (-freq[x], x)))

    2️⃣ Custom Bucket Sort¶

    # Example: Group numbers by their range
# LeetCode similar: "Maximum Gap"
def bucketSort(nums):
    if not nums:
        return []

    # Create buckets
    min_val, max_val = min(nums), max(nums)
    bucket_size = max(1, (max_val - min_val) // (len(nums) - 1))
    buckets = [[] for _ in range((max_val - min_val) // bucket_size + 1)]

    # Distribute numbers
    for num in nums:
        if num == max_val:
            bucket_idx = len(buckets) - 1
        else:
            bucket_idx = (num - min_val) // bucket_size
        buckets[bucket_idx].append(num)

    # Sort individual buckets
    result = []
    for bucket in buckets:
        if bucket:
            bucket.sort()
            result.extend(bucket)

    return result

    🚀 Advanced Techniques¶

    1️⃣ Priority Queue Pattern¶

    # Example: K-way merge with heap
# LeetCode similar: "Merge k Sorted Lists"
import heapq

def mergeKLists(lists):
    heap = []
    # Initialize heap with first elements
    for i, lst in enumerate(lists):
        if lst:
            heapq.heappush(heap, (lst[0], i, 0))  # (value, list_index, element_index)

    result = []
    while heap:
        val, list_idx, elem_idx = heapq.heappop(heap)
        result.append(val)
        # Push next element from same list
        if elem_idx + 1 < len(lists[list_idx]):
            heapq.heappush(heap, (lists[list_idx][elem_idx + 1], list_idx, elem_idx + 1))

    return result

    🎯 Interview Tips¶

    1. Always consider time/space complexity
    2. Look for opportunities to use custom sort keys
    3. Consider using tuples for multi-criteria sorting
    4. Remember built-in functions like sorted() and heapq

    🚀 Key Pattern Applications: - Multi-dimensional sorting → Use tuple as key - Frequency-based sorting → Use Counter + custom key - Complex object sorting → Create custom comparison - Large range numbers → Consider bucket sort - K-way merging → Use heap (priority queue)
