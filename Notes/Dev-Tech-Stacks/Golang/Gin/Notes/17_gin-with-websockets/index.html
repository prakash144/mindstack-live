    üìå Introduction¶

    WebSockets enable full-duplex, real-time communication between a client and a server. In Gin, we use the github.com/gorilla/websocket package to implement WebSockets.

    üîπ Installing Dependencies üì¶¶

    go get -u github.com/gorilla/websocket

    üîπ Setting Up a WebSocket Server üñ•Ô∏è¶

    package main

import (
    "net/http"
    "github.com/gin-gonic/gin"
    "github.com/gorilla/websocket"
)

var upgrader = websocket.Upgrader{
    CheckOrigin: func(r *http.Request) bool {
        return true // Allow all origins (modify for security)
    },
}

func handleWebSocket(c *gin.Context) {
    conn, err := upgrader.Upgrade(c.Writer, c.Request, nil)
    if err != nil {
        return
    }
    defer conn.Close()

    for {
        messageType, msg, err := conn.ReadMessage()
        if err != nil {
            break
        }
        conn.WriteMessage(messageType, msg) // Echo the message back
    }
}

func main() {
    r := gin.Default()
    r.GET("/ws", handleWebSocket)
    r.Run(":8080")
}

    console.log("Connected to server");
    socket.send("Hello, Server!");

    console.log("Message from server:", event.data);
};

socket.onerror = (error) => {
    console.log("WebSocket Error:", error);
};

socket.onclose = () => {
    console.log("Connection closed");
};

    üîπ Broadcasting Messages üì°¶

    type Client struct {
    conn *websocket.Conn
    send chan []byte
}

type Hub struct {
    clients    map[*Client]bool
    broadcast  chan []byte
    register   chan *Client
    unregister chan *Client
}

var hub = Hub{
    clients:    make(map[*Client]bool),
    broadcast:  make(chan []byte),
    register:   make(chan *Client),
    unregister: make(chan *Client),
}

func (h *Hub) run() {
    for {
        select {
        case client := <-h.register:
            h.clients[client] = true
        case client := <-h.unregister:
            if _, ok := h.clients[client]; ok {
                delete(h.clients, client)
                close(client.send)
            }
        case msg := <-h.broadcast:
            for client := range h.clients {
                select {
                case client.send <- msg:
                default:
                    close(client.send)
                    delete(h.clients, client)
                }
            }
        }
    }
}

func handleWebSocketBroadcast(c *gin.Context) {
    conn, _ := upgrader.Upgrade(c.Writer, c.Request, nil)
    client := &Client{conn: conn, send: make(chan []byte, 256)}
    hub.register <- client
    defer func() { hub.unregister <- client }()

    go func() {
        for msg := range client.send {
            client.conn.WriteMessage(websocket.TextMessage, msg)
        }
    }()

    for {
        _, msg, err := conn.ReadMessage()
        if err != nil {
            break
        }
        hub.broadcast <- msg
    }
}

func main() {
    go hub.run()
    r := gin.Default()
    r.GET("/ws", handleWebSocketBroadcast)
    r.Run(":8080")
}
