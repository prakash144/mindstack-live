    üèÜ Building RESTful APIs with Gin üöÄ¶

    üìå Introduction¶

    Gin is a high-performance web framework for Go, ideal for building RESTful APIs due to its simplicity, speed, and middleware support.

    üõ†Ô∏è Setting Up a Gin API¶

    package main

import (
    "github.com/gin-gonic/gin"
)

func main() {
    r := gin.Default() // Create a Gin router instance
    r.GET("/ping", func(c *gin.Context) {
        c.JSON(200, gin.H{"message": "pong"})
    })
    r.Run(":8080") // Run the server on port 8080
}

    üîπ Creating Handlers (Controllers) üéØ¶

    package controllers

import (
    "github.com/gin-gonic/gin"
    "net/http"
    "project/models"
)

var users []models.User

func GetUsers(c *gin.Context) {
    c.JSON(http.StatusOK, users)
}

func CreateUser(c *gin.Context) {
    var newUser models.User
    if err := c.ShouldBindJSON(&newUser); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }
    users = append(users, newUser)
    c.JSON(http.StatusCreated, newUser)
}

    üîπ Setting Up Routes üõ§Ô∏è¶

    package routes

import (
    "github.com/gin-gonic/gin"
    "project/controllers"
)

func RegisterRoutes(r *gin.Engine) {
    userRoutes := r.Group("/users")
    {
        userRoutes.GET("/", controllers.GetUsers)
        userRoutes.POST("/", controllers.CreateUser)
    }
}

    ‚û§ Custom Authentication Middleware¶

    func AuthMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        token := c.GetHeader("Authorization")
        if token == "" {
            c.JSON(http.StatusUnauthorized, gin.H{"error": "Unauthorized"})
            c.Abort()
            return
        }
        c.Next()
    }
}

    import (
    "gorm.io/driver/postgres"
    "gorm.io/gorm"
)

var DB *gorm.DB

func InitDB() {
    dsn := "host=localhost user=postgres password=yourpassword dbname=yourdb port=5432 sslmode=disable"
    var err error
    DB, err = gorm.Open(postgres.Open(dsn), &gorm.Config{})
    if err != nil {
        panic("Failed to connect to database!")
    }
    DB.AutoMigrate(&models.User{})
}
