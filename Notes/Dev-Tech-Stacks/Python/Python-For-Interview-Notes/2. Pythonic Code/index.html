    🔄 Unpacking¶

    📝 Basic Unpacking¶

    # List unpacking
point = [10, 20]
x, y = point  # x=10, y=20

# Tuple unpacking
coordinates = (5, 10, 15)
x, y, z = coordinates  # x=5, y=10, z=15

# String unpacking
first, *rest = "Python"  # first='P', rest=['y','t','h','o','n']

    🚫 Common Pitfalls¶

    # Too many values
x, y = [1, 2, 3]      # ValueError
# Too few values
x, y, z = [1, 2]      # ValueError

# Solution: Use * for variable length
first, *middle, last = [1, 2, 3, 4, 5]  # first=1, middle=[2,3,4], last=5

    🎯 Loop Operations¶

    📊 Loop Unpacking¶

    # Interview Pattern: Matrix Traversal
matrix = [[1, 2], [3, 4], [5, 6]]
for row, col in matrix:
    print(f"Row: {row}, Col: {col}")

# Interview Pattern: Pair Processing
pairs = [(1, 'one'), (2, 'two'), (3, 'three')]
for number, word in pairs:
    print(f"{number} is written as {word}")

    🔢 Enumerate Usage¶

    # Basic usage
fruits = ['apple', 'banana', 'cherry']
for i, fruit in enumerate(fruits):
    print(f"Index {i}: {fruit}")

# Start from different index
for i, fruit in enumerate(fruits, start=1):
    print(f"Fruit #{i}: {fruit}")

# Interview Pattern: Finding all indices
def find_all_indices(arr, target):
    return [i for i, num in enumerate(arr) if num == target]

    🔄 Zip Applications¶

    # Basic zip
keys = ['a', 'b', 'c']
values = [1, 2, 3]
dictionary = dict(zip(keys, values))  # {'a': 1, 'b': 2, 'c': 3}

# Multiple lists
names = ['Alice', 'Bob', 'Charlie']
ages = [25, 30, 35]
cities = ['NY', 'SF', 'LA']
for name, age, city in zip(names, ages, cities):
    print(f"{name} is {age} and lives in {city}")

# Interview Pattern: Parallel Processing
def compare_arrays(arr1, arr2):
    return [a == b for a, b in zip(arr1, arr2)]

    📊 Value Constraints¶

    # Boundary checking
def is_valid_age(age):
    return 0 <= age <= 120

# Value clamping
def clamp_value(value, min_val=0, max_val=100):
    return max(min_val, min(value, max_val))

# Interview Pattern: Matrix Bounds
def is_valid_cell(grid, row, col):
    return 0 <= row < len(grid) and 0 <= col < len(grid[0])

    🔧 Interview Tips & Tricks¶

    # Efficient min/max with multiple values
def get_bounds(numbers):
    return min(numbers), max(numbers)

# Multiple conditions
def is_valid_point(x, y):
    return 0 <= x < 100 and 0 <= y < 100

# Smart defaults with unpacking
def process_config(config=None):
    defaults = {'timeout': 30, 'retries': 3}
    return {**defaults, **(config or {})}

    🚀 Quick Reference for Interviews¶

    ✨ Common Operations¶

    # Sequence unpacking
first, *rest = [1, 2, 3, 4]  # first=1, rest=[2,3,4]

# Enumerate with offset
for i, val in enumerate(items, start=1):  # 1-based indexing

# Zip with default
for x, y in zip_longest(a, b, fillvalue=0):  # Handle uneven lists

# Multiple assignments
x, y = y, x  # Swap values

    ⚠️ Interview Gotchas¶

    • Always check for ValueError in unpacking
    • Remember zip stops at shortest list
    • Consider using enumerate when index is needed
    • Use * unpacking carefully with large sequences

    🎯 Key Interview Notes: - Unpacking makes code cleaner and more readable - Enumerate saves extra counter variables - Zip is perfect for parallel iteration - Use boundary checks for clean error handling - Remember Python's truthiness rules in conditions
