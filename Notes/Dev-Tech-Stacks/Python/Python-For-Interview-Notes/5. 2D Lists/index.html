    üìù Grid Creation¶

    # Creating a 2D grid
def create_grid(rows, cols, default_value=0):
    return [[default_value for _ in range(cols)] 
            for _ in range(rows)]

# Example usage
grid = create_grid(3, 4)  # 3x4 grid of zeros

# Initialize with values
matrix = [
    [1, 2, 3],
    [4, 5, 6],
    [7, 8, 9]
]

    üîß Grid Manipulation¶

    # Copy a grid
def deep_copy_grid(grid):
    return [row[:] for row in grid]

# Rotate a grid 90 degrees clockwise
def rotate_grid(grid):
    return list(zip(*grid[::-1]))

# Transpose a grid
def transpose_grid(grid):
    return list(map(list, zip(*grid)))

    ‚ö†Ô∏è Common Pitfalls¶

    # WRONG: Creates shared references

# RIGHT: Creates independent lists

# OR
grid = [[0] * 3 for _ in range(3)]

    üîÑ Matrix Traversal Patterns¶

    # DFS for grid
def dfs_grid(grid, i, j, visited=None):
    if visited is None:
        visited = set()

    if (i < 0 or i >= len(grid) or 
        j < 0 or j >= len(grid[0]) or 
        (i, j) in visited):
        return

    visited.add((i, j))

    # Visit all 4 directions
    directions = [(0,1), (1,0), (0,-1), (-1,0)]
    for di, dj in directions:
        dfs_grid(grid, i+di, j+dj, visited)

# BFS for grid
from collections import deque

def bfs_grid(grid, start_i, start_j):
    rows, cols = len(grid), len(grid[0])
    queue = deque([(start_i, start_j)])
    visited = {(start_i, start_j)}

    while queue:
        i, j = queue.popleft()

        for di, dj in [(0,1), (1,0), (0,-1), (-1,0)]:
            ni, nj = i + di, j + dj
            if (0 <= ni < rows and 0 <= nj < cols and 
                (ni, nj) not in visited):
                queue.append((ni, nj))
                visited.add((ni, nj))

    üìä Matrix Operations¶

    # Get diagonal elements
def get_diagonals(matrix):
    n = len(matrix)
    main_diag = [matrix[i][i] for i in range(n)]
    anti_diag = [matrix[i][n-1-i] for i in range(n)]
    return main_diag, anti_diag

# Check boundaries
def is_valid_cell(grid, i, j):
    return 0 <= i < len(grid) and 0 <= j < len(grid[0])

# Find neighbors
def get_neighbors(grid, i, j):
    directions = [(0,1), (1,0), (0,-1), (-1,0)]
    neighbors = []
    for di, dj in directions:
        ni, nj = i + di, j + dj
        if is_valid_cell(grid, ni, nj):
            neighbors.append((ni, nj))
    return neighbors

    üöÄ Time Complexity Guide¶

    # Common operations
Access: O(1)
Row iteration: O(n)
Full grid iteration: O(m*n)
Deep copy: O(m*n)
Rotation: O(m*n)

    ‚ö†Ô∏è Common Gotchas¶

    • Creating grids with shared references
    • Not checking boundaries
    • Confusing row/column ordering
    • Forgetting to mark visited cells in traversal
    • Not handling edge cases (empty grid, single cell)
