    🔄 Basic Operations¶

    📝 Dictionary Operations¶

    # Creation
hash_map = {}  # Empty dictionary
hash_map = dict()  # Alternative creation
hash_map = {"key": "value", "name": "John"}

# Basic Operations
hash_map["key"] = "value"     # Insert/Update
value = hash_map["key"]       # Access
del hash_map["key"]          # Delete
key_exists = "key" in hash_map  # Check existence

# Safe Operations
value = hash_map.get("key", "default")  # Safe access
value = hash_map.pop("key", "default")  # Safe delete

    🔍 Set Operations¶

    # Creation
hash_set = set()  # Empty set
hash_set = {1, 2, 3}  # Initialized set

# Basic Operations
hash_set.add(4)        # Insert
hash_set.remove(4)     # Delete (raises KeyError if missing)
hash_set.discard(4)    # Safe delete
exists = 4 in hash_set  # Check existence

    🔧 DefaultDict Usage¶

    from collections import defaultdict

# Count frequencies
freq = defaultdict(int)
for num in [1, 1, 2, 3, 3, 3]:
    freq[num] += 1  # No KeyError for new keys

# Group items
groups = defaultdict(list)
for key, val in pairs:
    groups[key].append(val)  # Automatically creates list

# Track indices
indices = defaultdict(set)
for i, num in enumerate(nums):
    indices[num].add(i)  # Track all positions of each number

    📊 Counter Class¶

    from collections import Counter

# Basic counting
nums = [1, 1, 2, 3, 3, 3]
count = Counter(nums)  # Counter({3: 3, 1: 2, 2: 1})

# Most common elements
top_k = count.most_common(2)  # [(3, 3), (1, 2)]

# Mathematical operations
c1 = Counter([1, 1, 2])
c2 = Counter([1, 2, 2])
print(c1 + c2)  # Add frequencies
print(c1 - c2)  # Subtract frequencies

    🔄 Two Sum Pattern¶

    def two_sum(nums, target):
    seen = {}  # val -> index
    for i, num in enumerate(nums):
        complement = target - num
        if complement in seen:
            return [seen[complement], i]
        seen[num] = i
    return []

    📊 Group Anagrams¶

    def group_anagrams(words):
    groups = defaultdict(list)
    for word in words:
        # Sort characters to get anagram key
        key = ''.join(sorted(word))
        groups[key].append(word)
    return list(groups.values())

    🎯 First Unique Character¶

    def first_unique_char(s):
    # Count frequencies
    count = Counter(s)

    # Find first char with count 1
    for i, char in enumerate(s):
        if count[char] == 1:
            return i
    return -1

    🚀 Time Complexity¶

    # Dictionary/Set Operations
Insert: O(1) average
Delete: O(1) average
Lookup: O(1) average
Iteration: O(n)

# Counter Operations
Creation: O(n)
most_common(): O(n log k) for top k elements

    ⚠️ Common Pitfalls¶

    # Mutable keys (will raise TypeError)
bad_dict = {[1, 2]: "value"}  # TypeError
bad_set = {[1, 2]}           # TypeError

# Safe key handling
good_dict = {tuple([1, 2]): "value"}  # Use tuple instead

# Missing key handling
# Bad:
value = dict_map["missing_key"]  # KeyError

# Good:
value = dict_map.get("missing_key", default)
if "missing_key" in dict_map:
    value = dict_map["missing_key"]
