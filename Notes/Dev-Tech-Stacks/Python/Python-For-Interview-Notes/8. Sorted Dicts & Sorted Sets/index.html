    üîÑ SortedDict Operations¶

    üìù Basic Operations¶

    from sortedcontainers import SortedDict

# Creation
sd = SortedDict()
sd = SortedDict({'c': 3, 'a': 1, 'b': 2})  # Auto-sorts keys

# Insertion
sd['d'] = 4
sd.update({'e': 5, 'f': 6})

# Access
value = sd['a']  # Direct access
value = sd.get('x', default=0)  # Safe access

# Deletion
del sd['a']  # Remove by key
first = sd.popitem(0)  # Remove first item
last = sd.popitem()    # Remove last item

    üîç Advanced Operations¶

    # Get index of key
index = sd.index('c')

# Get key at index
key = sd.peekitem(index)

# Slicing
subset = sd.items()[1:4]  # Get slice of items

# Binary search operations
index = sd.bisect_left(key)
index = sd.bisect_right(key)

    ‚öôÔ∏è SortedSet Operations¶

    üìù Basic Operations¶

    from sortedcontainers import SortedSet

# Creation
ss = SortedSet()
ss = SortedSet([3, 1, 4, 1, 5])  # Duplicates removed

# Insertion
ss.add(2)
ss.update([6, 7, 8])

# Removal
ss.remove(2)          # Raises KeyError if missing
ss.discard(2)         # No error if missing
first = ss.pop(0)     # Remove first element
last = ss.pop()       # Remove last element

    üîç Advanced Operations¶

    # Binary search operations
index = ss.bisect_left(value)
index = ss.bisect_right(value)

# Range operations
subset = ss.irange(start, end)
subset = ss.irange(start, end, reverse=True)

# Count elements in range
count = ss.count(start, end)

    üîÑ Range-Based Operations¶

    def find_range(sorted_set, start, end):
    # Get all elements in range [start, end]
    return sorted_set.irange(start, end)

def count_in_range(sorted_set, start, end):
    # Count elements in range [start, end]
    return sorted_set.count(start, end)

    üìä Maintaining Sorted Order¶

    def track_scores(scores):
    # Keep scores sorted with O(log n) updates
    sorted_scores = SortedDict()

    def add_score(player, score):
        sorted_scores[player] = score

    def get_rank(player):
        return sorted_scores.index(player) + 1

    def get_top_k(k):
        return dict(list(sorted_scores.items())[-k:])

    üéØ Finding Closest Elements¶

    def find_closest(sorted_set, target):
    # Find element closest to target
    index = sorted_set.bisect_left(target)

    if index == 0:
        return sorted_set[0]
    if index == len(sorted_set):
        return sorted_set[-1]

    before = sorted_set[index - 1]
    after = sorted_set[index]

    return before if target - before < after - target else after

    üöÄ Time Complexity¶

    # SortedDict/SortedSet Operations
Insert/Delete: O(log n)
Lookup: O(log n)
Index access: O(log n)
Iteration: O(n)
Slice: O(k) where k is slice size

    ‚ö†Ô∏è Common Pitfalls¶

    # Modifying during iteration
# Wrong:
for key in sorted_dict:
    if condition:
        del sorted_dict[key]  # May raise error

# Right:
to_delete = [k for k in sorted_dict if condition]
for key in to_delete:
    del sorted_dict[key]

# Comparing with regular dict/set
# SortedDict/SortedSet maintain order but are slower
# Use when order matters more than raw performance
